Java
class Solution {
    public int minFlips(int a, int b, int c) {
        int res = 0;
        for (int i = 0; i < 32; i++) {
            int num_a = a & 1; a >>= 1; // е°ҶжңҖеҗҺдёҖдҪҚеҸ–еҮә
            int num_b = b & 1; b >>= 1; 
            int num_c = c & 1; c >>= 1;
            if(num_a == 0 && num_b == 0 && num_c == 1) res += 1;
            else if(num_a == 1 && num_b == 1 && num_c == 0) res += 2;
            else if(num_c == 0 && num_a + num_b == 1) res += 1;
        }
        return res;
    }
}
