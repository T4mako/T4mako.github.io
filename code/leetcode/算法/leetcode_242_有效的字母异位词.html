242. 有效的字母异位词

题目描述

class Solution {
    public boolean isAnagram(String s, String t) {
		if(s.length() != t.length()) return false;
		int len = s.length();
		int[] num1 = new int[26];
		int[] num2 = new int[26];
		for(int i = 0;i < len;i++) {
			num1[s.charAt(i) - 97]++;
			num2[t.charAt(i) - 97]++;
		}
		for(int i = 0;i < 26;i++) {
			if(num1[i] != num2[i]) return false;
		}
		return true;
    }
}
