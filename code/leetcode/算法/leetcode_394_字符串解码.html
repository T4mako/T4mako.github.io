class Solution {
    public String decodeString(String s) {
        StringBuilder res = new StringBuilder();
        ArrayList<Object> list = new ArrayList<>();
        for (int i = 0; i < s.length(); i++) {
            if(s.charAt(i) >= 97){
                // йҒҚеҺҶеҲ°еӯ—жҜҚпјҢе…Ҙж Ҳ
                list.add(s.charAt(i));
            }else if(s.charAt(i) < 58){
                // йҒҚеҺҶеҲ°ж•°еӯ— е°Ҷж•ҙдёӘж•°еӯ—е…Ҙж Ҳ
                list.add(Integer.valueOf(s.substring(i,s.indexOf('[',i))));
                i = s.indexOf('[',i) - 1;
            }else if(s.charAt(i) == '['){
                // йҒҚеҺҶеҲ° '[' пјҢе…Ҙж Ҳ
                list.add('[');
            } else {
                // йҒҚеҺҶеҲ° ']'
                StringBuilder temp = new StringBuilder();
                int j = list.size() - 1;
                for (; j >= 0 ; j--) {
                    if(!"[".equals(list.get(j).toString())){
                        temp.insert(0, list.get(j));
                        list.remove(j);
                    }else {break;}
                }
                int num = (Integer) list.get(--j);
                StringBuilder str = new StringBuilder();
                for (int k = 0; k < num; k++) {
                    str.append(temp);
                }
                list.set(j,str);
                list.remove(j+1);
            }
        }
        for (Object o : list) {
            res.append(o);
        }
        return res.toString();
    }
}
