public class PhoneCode {

    //1гҖҒз”ҹжҲҗ6дёәж•°еӯ—йӘҢиҜҒз Ғ
    public static String getCode(){
        Random random = new Random();
        int code = 0;
        for (int i = 0; i < 6; i++) {
            code *= 10;
            code += random.nextInt(10);
        }
        StringBuffer stringBuffer = new StringBuffer();
        stringBuffer.append(code);
        return stringBuffer.toString();
    }

    //2гҖҒжҜҸдёӘжүӢжңәеҸӘиғҪеҸ‘йҖҒдёүж¬ЎпјҢйӘҢиҜҒз Ғж”ҫеҲ°redisдёӯпјҢи®ҫзҪ®иҝҮжңҹж—¶й—ҙ
    public static void verifyCode(String phone){
        //иҝһжҺҘredis
        Jedis jedis = new Jedis("192.168.80.130", 6379);

        //жӢјжҺҘkey
        //жүӢжңәеҸ‘йҖҒж¬Ўж•°key
        String countKey = "VerifyCode" + phone + ":count";
        //йӘҢиҜҒз Ғkey
        String codeKey = "VerfyCode" + phone + ":code";

        //жҜҸдёӘжүӢжңәеҸӘиғҪеҸ‘йҖҒдёүж¬Ў
        String count = jedis.get(codeKey);
        if(count == null){
            //жІЎжңүеҸ‘йҖҒж¬Ўж•°пјҢз¬¬дёҖж¬ЎеҸ‘йҖҒ
            //и®ҫзҪ®еҸ‘йҖҒж¬Ўж•°жҳҜ1
            jedis.setex(countKey,24*60*60,"1");
        }else if (Integer.parseInt(count) <= 2){
            //еҸ‘йҖҒж¬Ўж•°+1
            jedis.incr(countKey);
        }else {
            System.out.println("д»ҠеӨ©зҡ„еҸ‘йҖҒж¬Ўж•°е·Іи¶…иҝҮдёүж¬Ў");
            jedis.close();
            return;
        }

        //еҸ‘йҖҒзҡ„йӘҢиҜҒз Ғж”ҫеҲ°redisйҮҢ
        String vcode = getCode();
        jedis.setex(codeKey,120,vcode);
        jedis.close();

        jedis.close();
    }

    //3гҖҒйӘҢиҜҒз Ғж ЎйӘҢ

        }
        jedis.close();
    }
}

//ејҖеҗҜзј“еӯҳ
@EnableCaching
//й…ҚзҪ®зұ»
@Configuration
public class RedisConfig extends CachingConfigurerSupport {
    @Bean
    public RedisTemplate<String, Object> redisTemplate(RedisConnectionFactory factory) {
        RedisTemplate<String, Object> template = new RedisTemplate<>();
        RedisSerializer<String> redisSerializer = new StringRedisSerializer();
        Jackson2JsonRedisSerializer jackson2JsonRedisSerializer = new Jackson2JsonRedisSerializer(Object.class);
        ObjectMapper om = new ObjectMapper();
        om.setVisibility(PropertyAccessor.ALL, JsonAutoDetect.Visibility.ANY);
        om.enableDefaultTyping(ObjectMapper.DefaultTyping.NON_FINAL);
        jackson2JsonRedisSerializer.setObjectMapper(om);
        template.setConnectionFactory(factory);
        //keyеәҸеҲ—еҢ–ж–№ејҸ
        template.setKeySerializer(redisSerializer);
        //valueеәҸеҲ—еҢ–
        template.setValueSerializer(jackson2JsonRedisSerializer);
        //value hashmapеәҸеҲ—еҢ–
        template.setHashValueSerializer(jackson2JsonRedisSerializer);
        return template;
    }

    @Bean
    public CacheManager cacheManager(RedisConnectionFactory factory) {
        RedisSerializer<String> redisSerializer = new StringRedisSerializer();
        Jackson2JsonRedisSerializer jackson2JsonRedisSerializer = new Jackson2JsonRedisSerializer(Object.class);
        //и§ЈеҶіжҹҘиҜўзј“еӯҳиҪ¬жҚўејӮеёёзҡ„й—®йўҳ
        ObjectMapper om = new ObjectMapper();
        om.setVisibility(PropertyAccessor.ALL, JsonAutoDetect.Visibility.ANY);
        om.enableDefaultTyping(ObjectMapper.DefaultTyping.NON_FINAL);
        jackson2JsonRedisSerializer.setObjectMapper(om);
        // й…ҚзҪ®еәҸеҲ—еҢ–пјҲи§ЈеҶід№ұз Ғзҡ„й—®йўҳпјү,иҝҮжңҹж—¶й—ҙ600з§’
        RedisCacheConfiguration config = RedisCacheConfiguration.defaultCacheConfig()
                .entryTtl(Duration.ofSeconds(600))
                .serializeKeysWith(RedisSerializationContext.SerializationPair.fromSerializer(redisSerializer))
                .serializeValuesWith(RedisSerializationContext.SerializationPair.fromSerializer(jackson2JsonRedisSerializer))
                .disableCachingNullValues();
        RedisCacheManager cacheManager = RedisCacheManager.builder(factory)
                .cacheDefaults(config)
                .build();
        return cacheManager;
    }
}

@GetMapping("testLock")
public void testLock(){

    Boolean lock = redisTemplate.opsForValue().setIfAbsent("lock", uuid,3, TimeUnit.SECONDS);
    //2иҺ·еҸ–й”ҒжҲҗеҠҹгҖҒжҹҘиҜўnumзҡ„еҖј
    if(lock){
        Object value = redisTemplate.opsForValue().get("num");
        //2.1еҲӨж–ӯnumдёәз©әreturn
        if(StringUtils.isEmpty(value)){
            return;
        }
        //2.2жңүеҖје°ұиҪ¬жҲҗжҲҗint
        int num = Integer.parseInt(value+"");
        //2.3жҠҠredisзҡ„numеҠ 1
        redisTemplate.opsForValue().set("num", ++num);
        //2.4йҮҠж”ҫй”ҒпјҢdel
        //жҜ”иҫғuuidеҖјжҳҜеҗҰдёҖж ·
        String lockUuid = (String) redisTemplate.opsForValue().get("lock");
        if(lockUuid.equals(uuid)){
            redisTemplate.delete("lock");
        }
    }else{
        //3иҺ·еҸ–й”ҒеӨұиҙҘгҖҒжҜҸйҡ”0.1з§’еҶҚиҺ·еҸ–
        try {
            Thread.sleep(100);
            testLock();
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }
}

@GetMapping("testLockLua")
public void testLockLua() {
    //1 еЈ°жҳҺдёҖдёӘuuid ,е°ҶеҒҡдёәдёҖдёӘvalue ж”ҫе…ҘжҲ‘д»¬зҡ„keyжүҖеҜ№еә”зҡ„еҖјдёӯ
    String uuid = UUID.randomUUID().toString();
    //2 е®ҡд№үдёҖдёӘй”Ғпјҡlua и„ҡжң¬еҸҜд»ҘдҪҝз”ЁеҗҢдёҖжҠҠй”ҒпјҢжқҘе®һзҺ°еҲ йҷӨпјҒ
    String skuId = "25"; // и®ҝй—®skuId дёә25еҸ·зҡ„е•Ҷе“Ғ 100008348542
    String locKey = "lock:" + skuId; // й”ҒдҪҸзҡ„жҳҜжҜҸдёӘе•Ҷе“Ғзҡ„ж•°жҚ®

    // 3 иҺ·еҸ–й”Ғ
    Boolean lock = redisTemplate.opsForValue().setIfAbsent(locKey, uuid, 3, TimeUnit.SECONDS);

    // з¬¬дёҖз§Қпјҡ lock дёҺиҝҮжңҹж—¶й—ҙдёӯй—ҙдёҚеҶҷд»»дҪ•зҡ„д»Јз ҒгҖӮ
    // redisTemplate.expire("lock",10, TimeUnit.SECONDS);//и®ҫзҪ®иҝҮжңҹж—¶й—ҙ
    // еҰӮжһңtrue
    if (lock) {
        // жү§иЎҢзҡ„дёҡеҠЎйҖ»иҫ‘ејҖе§Ӣ
        // иҺ·еҸ–зј“еӯҳдёӯзҡ„num ж•°жҚ®
        Object value = redisTemplate.opsForValue().get("num");
        // еҰӮжһңжҳҜз©әзӣҙжҺҘиҝ”еӣһ
        if (StringUtils.isEmpty(value)) {
            return;
        }
        // дёҚжҳҜз©ә еҰӮжһңиҜҙеңЁиҝҷеҮәзҺ°дәҶејӮеёёпјҒ йӮЈд№Ҳdelete е°ұеҲ йҷӨеӨұиҙҘпјҒ д№ҹе°ұжҳҜиҜҙй”Ғж°ёиҝңеӯҳеңЁпјҒ
        int num = Integer.parseInt(value + "");
        // дҪҝnum жҜҸж¬Ў+1 ж”ҫе…Ҙзј“еӯҳ
        redisTemplate.opsForValue().set("num", String.valueOf(++num));
        /*дҪҝз”Ёluaи„ҡжң¬жқҘй”Ғ*/
        // е®ҡд№үlua и„ҡжң¬
        String script = "if redis.call('get', KEYS[1]) == ARGV[1] then return redis.call('del', KEYS[1]) else return 0 end";
        // дҪҝз”Ёredisжү§иЎҢluaжү§иЎҢ
        DefaultRedisScript<Long> redisScript = new DefaultRedisScript<>();
        redisScript.setScriptText(script);
        // и®ҫзҪ®дёҖдёӢиҝ”еӣһеҖјзұ»еһӢ дёәLong
        // еӣ дёәеҲ йҷӨеҲӨж–ӯзҡ„ж—¶еҖҷпјҢиҝ”еӣһзҡ„0,з»ҷе…¶е°ҒиЈ…дёәж•°жҚ®зұ»еһӢгҖӮеҰӮжһңдёҚе°ҒиЈ…йӮЈд№Ҳй»ҳи®Өиҝ”еӣһString зұ»еһӢпјҢ
        // йӮЈд№Ҳиҝ”еӣһеӯ—з¬ҰдёІдёҺ0 дјҡжңүеҸ‘з”ҹй”ҷиҜҜгҖӮ
        redisScript.setResultType(Long.class);
        // з¬¬дёҖдёӘиҰҒжҳҜscript и„ҡжң¬ пјҢз¬¬дәҢдёӘйңҖиҰҒеҲӨж–ӯзҡ„keyпјҢз¬¬дёүдёӘе°ұжҳҜkeyжүҖеҜ№еә”зҡ„еҖјгҖӮ
        redisTemplate.execute(redisScript, Arrays.asList(locKey), uuid);
    } else {
        // е…¶д»–зәҝзЁӢзӯүеҫ…

}
